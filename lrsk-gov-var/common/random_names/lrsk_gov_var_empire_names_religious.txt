### Democratic Spiritualist

# Democratic Spiritualist 1
empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 1
			OR = {
				has_government = "lrsk_gov_var_gov_cosmic_choir"
				has_government = "lrsk_gov_var_gov_saintly_assembly"
			}
			is_pirate = no
			is_primitive = no
			NOT = { is_country_type = fallen_empire }
			NOT = { is_country_type = awakened_fallen_empire }
		}
	}
	format = "format.dem_spi.1"
	noun = "[This.Capital.System.GetName]"
	prefix_format = "[This.Capital.System.GetName] <democratic_spi>"
	# Empire of Sol
}
