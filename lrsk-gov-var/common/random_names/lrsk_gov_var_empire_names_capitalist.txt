### Democratic Generic

# Democratic Generic 1
empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 1
			OR = {
				has_government = "lrsk_gov_var_gov_libertarian_freehold"
				has_government = "lrsk_gov_var_gov_enterprise_society"
			}
			is_pirate = no
			is_primitive = no
			NOT = { is_country_type = fallen_empire }
			NOT = { is_country_type = awakened_fallen_empire }
		}
	}
	format = "format.dem_gen.1"
	noun = "[This.Capital.System.GetName]"
	prefix_format = "[This.Capital.System.GetName] <democratic_gen>"
	# Empire of Sol
}
