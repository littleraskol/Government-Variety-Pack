### Imperial Generic

# Imperial Generic 1
empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 1
			OR = {
				has_government = "lrsk_gov_var_gov_feudal_empire"
				has_government = "lrsk_gov_var_gov_absolute_monarchy"
				has_government = "lrsk_gov_var_gov_constitutional_monarchy"
				has_government = "lrsk_gov_var_gov_revolutionary_empire"
				has_government = "lrsk_gov_var_gov_solar_monarchy"
				has_government = "lrsk_gov_var_gov_libertine_empire"
			}
			is_pirate = no
			is_primitive = no
			NOT = { is_country_type = fallen_empire }
			NOT = { is_country_type = awakened_fallen_empire }
		}
	}
	format = "format.imp_gen.1"
	noun = "[This.Capital.System.GetName]"
	prefix_format = "[This.Capital.System.GetName] <imperial_gen>"
	# Empire of Sol
}
