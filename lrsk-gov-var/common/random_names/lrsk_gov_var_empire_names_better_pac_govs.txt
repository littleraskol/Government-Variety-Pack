### Democratic Pacifist

# Democratic Pacifist 1
empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 25
			OR = {
				has_government = "lrsk_gvp_gov_tranquil_republic"
				has_government = "lrsk_gvp_gov_tranquil_commonwealth"
			}
			is_pirate = no
			is_primitive = no
			NOT = { is_country_type = fallen_empire }
			NOT = { is_country_type = awakened_fallen_empire }
		}
	}
	format = "{<lrsk_pac_desc>{AofB{<lrsk_democratic_pac> [This.Capital.System.GetName]}}}"
	noun = "{[This.Capital.System.GetName]}"
	prefix_format = "{AofBpfx{[This.Capital.System.GetName] <lrsk_democratic_pac>}}"
}
