## This file defines random names for NPC-only governments required for other content.
## Nothing should be player-facing, thus preserving patch fucntions.
## iow, "lesser names patch" will miss this file.

### Imperial Generic

# Imperial Generic 1
empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 15
			OR = {
				has_government = "lrsk_gov_var_gov_colonial_viceroyalty_imp"
			}
			is_pirate = no
			is_primitive = no
			NOT = { is_country_type = fallen_empire }
			NOT = { is_country_type = awakened_fallen_empire }
		}
	}
	
	lookups = "<lrsk_impdic_filler> <lrsk_imperial_gen> [This.Capital.System.GetName]"
	format = format.lrsk_imp_gen.1
	noun = format.home_system
	prefix_format = format_prefix.lrsk_imp_gen.1
	# Empire of Sol
}
