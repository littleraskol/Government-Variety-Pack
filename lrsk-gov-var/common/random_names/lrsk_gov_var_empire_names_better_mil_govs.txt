### Imperial Militarist

# Imperial Militarist 1
empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 25
			OR = {
				has_government = "lrsk_gov_var_gov_galactic_imperium"
				has_government = "lrsk_gov_var_gov_crusader_kingdom"
				has_government = "lrsk_gov_var_gov_folkish_realm"
			}
			is_pirate = no
			is_primitive = no
			NOT = { is_country_type = fallen_empire }
			NOT = { is_country_type = awakened_fallen_empire }
		}
	}
	
	lookups = "<lrsk_mil_desc> <lrsk_imperial_mil> [This.Capital.System.GetName]"
	format = format.lrsk_imp_mil.1
	noun = format.home_system
	prefix_format = format_prefix.lrsk_imp_mil.1
	# Empire of Sol
