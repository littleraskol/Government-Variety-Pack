# Megacorporation
empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 20
			OR = {
				has_government = "lrsk_gov_var_gov_galactic_cartel"
				has_government = "lrsk_gov_var_gov_proprietary_franchise"
				#has_government = "lrsk_gov_var_gov_colonial_viceroyalty_cmp"
			}
			is_pirate = no
			is_primitive = no
			NOT = { is_country_type = fallen_empire }
			NOT = { is_country_type = awakened_fallen_empire }
		}
	}
	
	lookups = "[This.GetSpeciesAdj] <lrsk_stellar_filler_desc> <lrsk_megacorp> <lrsk_corp_bits> [This.GetSpeciesName]"
	format = format.lrsk_megacorp.1
	noun = format.species
	prefix_format = format_prefix.lrsk_meg_gen.3
	# Human Empire
}
