# Academy States

empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 20
			OR = {
				has_government = "lrsk_gov_var_gov_academy_state"
			}
			is_pirate = no
			is_primitive = no
			NOT = { is_country_type = fallen_empire }
			NOT = { is_country_type = awakened_fallen_empire }
		}
	}
	
	lookups = "<lrsk_schl_desc> <lrsk_oligarchic_mat> [This.Capital.System.GetName]"
	format = format.lrsk_acst.1
	noun = format.home_system
	prefix_format = format_prefix.lrsk_oli_mat.1
	# Sol Empire
}
