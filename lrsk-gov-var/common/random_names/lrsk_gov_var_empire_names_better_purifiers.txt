empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 20
			OR = {
				has_government = "lrsk_gov_var_gov_doom_horde"
				has_government = "lrsk_gov_var_gov_doom_horde_imp"
			}
			is_pirate = no
			is_primitive = no
			NOT = { is_country_type = fallen_empire }
			NOT = { is_country_type = awakened_fallen_empire }
		}
	}
	
	lookups = "[This.GetSpeciesAdj] <lrsk_hrd_pur_desc> <lrsk_hrd_purifiers>"
	format = "format.lrsk_hrd_purifier.1"
	noun = format.species
	prefix_format = format_prefix.lrsk_hrd_pur.3
	# Human Empire
}
