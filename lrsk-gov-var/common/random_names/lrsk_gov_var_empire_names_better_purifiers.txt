## Fanatical Purifiers

# Imperial

# Fanatical Purifiers
empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 20
			OR = {
				has_government = "gov_blood_court"
			}
			is_pirate = no
			is_primitive = no
			NOT = { is_country_type = fallen_empire }
			NOT = { is_country_type = awakened_fallen_empire }
		}
	}
	format = "format.lrsk_imp_purifier.1"
	noun = "[This.GetSpeciesName]"
	prefix_format = "[This.GetSpeciesName] <lrsk_imp_purifiers>"
	# Human Empire
}
