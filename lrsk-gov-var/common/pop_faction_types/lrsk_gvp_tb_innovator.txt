	on_create = {
		root.owner = {
			set_country_flag = lrsk_gvp_tb_innovators_arose
			# We want to weaken and render obsolete the vanilla materialist faction
			if = {
				limit = { has_faction = technologist }
				every_owned_pop = { limit = { member_of_faction = technologist }
					set_pop_faction = gvp_tb_innovator
				}
				
				if = {
					limit = {
						any_owned_leader = { leader_of_faction = technologist }
						not = { ruler = { leader_of_faction = gvp_tb_innovator } } # worth preserving
					}
					random_owned_leader = { limit = { leader_of_faction = technologist }
						save_event_target_as = lrsk_gvp_tb_former_techfac_leader
					}
					root = { assign_leader = event_target:lrsk_gvp_tb_former_techfac_leader }
				}
			}
		}
		pop_faction_event = { id = factions.1 } # call after to make sure leader assignment makes sense
	}
