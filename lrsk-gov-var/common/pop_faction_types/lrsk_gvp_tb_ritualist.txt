gvp_tb_ritualist = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_spiritualist

	unique = yes
	support_multiplier = 1

	resources = {
		category = pop_factions
		produces = {
			influence = 2
		}
	}

	is_potential = {
		or = {
			has_civic = lrsk_adv_civ_civic_technobarbarism
			has_modifier = lrsk_adv_civ_tb_revolution
		}
		
		# Ritualists are literally reactionaries
		root.owner = { OR = {
			has_faction = gvp_tb_innovator
			NOT = { has_policy_flag = lrsk_adv_civ_full_rev }
		} }
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no
		is_robot_pop = no
	}

	can_join_faction = {
		OR = {
			has_ethic = ethic_spiritualist
			has_ethic = ethic_authoritarian
		}
		NOR = {
			has_trait = trait_nerve_stapled
			has_ethic = ethic_materialist
			has_ethic = ethic_egalitarian
		}
	}

	attraction = {
		base = 300 # want to avoid the vanilla version getting pull as much as possible...

		modifier = {
			factor = 1.3
			exists = from
			from = {
				exists = leader
				leader = { leader_class = ruler }
			}
		}

		modifier = {
			factor = 0.8
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}
	}

	leader = {
		base = 100
		modifier = {
			factor = 0
			species = {
				has_trait = "trait_mechanical"
			}
		}

		modifier = {
			factor = 0.0
			is_event_leader = yes
		}
		
		# dunno if this will work but we want to weaken the vanilla materialist faction however possible
		modifier = {
			factor = 3
			leader_of_faction = traditionalist
		}
	}
	
	demand = {
		title = "GVP_TB_RITUALIST_FULLREV"
		unfulfilled_title = "GVP_TB_RITUALIST_NO_FULLREV"
		desc = "GVP_TB_RITUALIST_FULLREV_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = 0

		potential = {
			exists = owner
			owner = { or = {
				has_policy_flag = lrsk_adv_civ_full_rev
				has_policy_flag = lrsk_adv_civ_lapsed_rev
			} }
		}

		trigger = {
			owner = {
				has_policy_flag = lrsk_adv_civ_full_rev
			}
		}
	}
	
	demand = {
		title = "GVP_TB_RITUALIST_LAPSEREV"
		unfulfilled_title = "GVP_TB_RITUALIST_NO_LAPSEREV"
		desc = "GVP_TB_RITUALIST_LAPSEREV_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = 0

		potential = {
			exists = owner
			owner = { or = {
				has_policy_flag = lrsk_adv_civ_lapsed_rev
				has_policy_flag = lrsk_adv_civ_token_rev
			} }
		}

		trigger = {
			owner = {
				has_policy_flag = lrsk_adv_civ_lapsed_rev
			}
		}
	}
	
	demand = {
		title = "GVP_TB_RITUALIST_TOKENREV"
		unfulfilled_title = "GVP_TB_RITUALIST_NO_TOKENREV"
		desc = "GVP_TB_RITUALIST_TOKENREV_DESC"

		fulfilled_effect = 0
		unfulfilled_effect = -5

		potential = {
			exists = owner
			owner = { or = {
				has_policy_flag = lrsk_adv_civ_token_rev
				has_policy_flag = lrsk_adv_civ_no_rev
			} }
		}

		trigger = {
			owner = {
				not = { has_policy_flag = lrsk_adv_civ_token_rev }
			}
		}
	}
	
	demand = {
		title = "GVP_TB_RITUALIST_UNREV"
		unfulfilled_title = "GVP_TB_RITUALIST_NO_UNREV"
		desc = "GVP_TB_RITUALIST_UNREV_DESC"

		fulfilled_effect = 0
		unfulfilled_effect = -10

		potential = {
			exists = owner
			owner = {
				has_policy_flag = lrsk_adv_civ_no_rev
			}
		}

		trigger = {
			owner = {
				not = { has_policy_flag = lrsk_adv_civ_no_rev }
			}
		}
	}

	demand = {
		title = "GVP_TB_RITUALIST_AUTHORITY"
		unfulfilled_title = "GVP_TB_RITUALIST_NO_AUTHORITY"
		desc = "GVP_TB_RITUALIST_AUTHORITY_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = -10

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				OR = {
					has_ethic = ethic_fanatic_authoritarian
					has_ethic = ethic_authoritarian
				}
			}
		}
	}

	demand = {
		title = "GVP_TB_RITUALIST_SPIRITUALISM"
		unfulfilled_title = "GVP_TB_RITUALIST_NO_SPIRITUALISM"
		desc = "GVP_TB_RITUALIST_DISCOVERY_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -20

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				OR = {
					has_ethic = ethic_fanatic_spiritualist
					has_ethic = ethic_spiritualist
				}
			}
		}
	}

	demand = {
		title = "GVP_TB_RITUALIST_SPIRITUALIST_POPS"
		unfulfilled_title = "GVP_TB_RITUALIST_FEW_SPIRITUALIST_POPS"
		desc = "GVP_TB_RITUALIST_SPIRITUALIST_POPS_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -0.001

		potential = {
			support >= 0.2
		}

		trigger = {
			support >= 0.4
		}
	}

	demand = {
		title = "GVP_TB_RITUALIST_UNSPIRITUALIST"
		unfulfilled_title = "GVP_TB_RITUALIST_NOT_UNSPIRITUALIST"
		desc = "GVP_TB_RITUALIST_UNSPIRITUALIST_DESC"

		unfulfilled_effect = -10
		fulfilled_effect = 0.001

		potential = {
			support < 0.4
		}

		trigger = {
			support >= 0.2
		}
	}

	demand = {
		title = "GVP_TB_RITUALIST_TOMB_WORLDS"
		unfulfilled_title = "GVP_TB_RITUALIST_NO_TOMB_WORLDS"
		desc = "GVP_TB_RITUALIST_TOMB_WORLDS_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				has_surveyed_class = pc_nuked
				has_ascension_perk = ap_consecrated_worlds
			}
		}

		trigger = {
			owner = {
				any_planet_within_border = {
					is_planet_class = pc_nuked
					is_colony = no
					OR = {
						has_modifier = consecrated_quality_1
						has_modifier = consecrated_quality_2
						has_modifier = consecrated_quality_3
						has_modifier = consecrated_quality_4
					}
				}
			}
		}
	}

	demand = {
		title = "GVP_TB_RITUALIST_RELIC_WORLDS"
		unfulfilled_title = "GVP_TB_RITUALIST_NO_RELIC_WORLDS"
		desc = "GVP_TB_RITUALIST_RELIC_WORLDS_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				has_surveyed_class = pc_relic
				has_ascension_perk = ap_consecrated_worlds
			}
		}

		trigger = {
			owner = {
				any_planet_within_border = {
					is_planet_class = pc_relic
					is_colony = no
					OR = {
						has_modifier = consecrated_quality_1
						has_modifier = consecrated_quality_2
						has_modifier = consecrated_quality_3
						has_modifier = consecrated_quality_4
					}
				}
			}
		}
	}

	demand = { # "GVP_TB_RITUALIST_SHROUD_BOON"
		title = "GVP_TB_RITUALIST_SHROUD_BOON"
		unfulfilled_title = "GVP_TB_RITUALIST_NO_SHROUD_BOON"
		desc = "GVP_TB_RITUALIST_SHROUD_BOON_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				has_country_flag = positive_shroud_outcome
			}
		}

		trigger = {
			owner = {
				has_country_flag = positive_shroud_outcome
			}
		}
	}

	demand = { # "GVP_TB_RITUALIST_SHROUD_CURSE"
		title = "GVP_TB_RITUALIST_SHROUD_CURSE"
		unfulfilled_title = "GVP_TB_RITUALIST_NO_SHROUD_CURSE"
		desc = "GVP_TB_RITUALIST_SHROUD_CURSE_DESC"

		fulfilled_effect = -5
		unfulfilled_effect = 0.001

		potential = {
			exists = owner
			owner = {
				has_country_flag = negative_shroud_outcome
			}
		}

		trigger = {
			owner = {
				has_country_flag = negative_shroud_outcome
			}
		}
	}

	demand = {
		title = "GVP_TB_RITUALIST_RELIGIOUS_FINDING"
		unfulfilled_title = "GVP_TB_RITUALIST_NO_RELIGIOUS_FINDING"
		desc = "GVP_TB_RITUALIST_RELIGIOUS_FINDING_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				has_modifier = proclaim_religious_finding
			}
		}

		trigger = {
			owner = {
				has_modifier = proclaim_religious_finding
			}
		}
	}

	demand = {
		title = "GVP_TB_RITUALIST_STRATIFICATION"
		unfulfilled_title = "GVP_TB_RITUALIST_NO_STRATIFICATION"
		desc = "GVP_TB_RITUALIST_STRATIFICATION_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -5

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				OR = {
					any_owned_species = {
						has_living_standard = { type = living_standard_stratified country = prev }
						has_living_standard = { type = living_standard_academic_privilege country = prev }
						has_citizenship_type = { type = citizenship_robot_servitude country = prev }
					}
				}
			}
		}
	}

	demand = {
		title = "GVP_TB_RITUALIST_SLAVERY"
		unfulfilled_title = "GVP_TB_RITUALIST_NO_SLAVERY"
		desc = "GVP_TB_RITUALIST_SLAVERY_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = 0

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				has_policy_flag = slavery_allowed
			}
		}
	}
	
	on_create = {
		root.owner = {
			# We want to weaken and render obsolete the vanilla spiritualist faction
			if = {
				limit = { has_faction = traditionalist }
				every_owned_pop = { limit = { member_of_faction = traditionalist }
					set_pop_faction = gvp_tb_ritualist
				}
				
				if = {
					limit = {
						any_owned_leader = { leader_of_faction = traditionalist }
						not = { ruler = { leader_of_faction = gvp_tb_ritualist } } # worth preserving
					}
					random_owned_leader = { limit = { leader_of_faction = traditionalist }
						save_event_target_as = lrsk_gvp_tb_former_tradfac_leader
					}
					root = { assign_leader = event_target:lrsk_gvp_tb_former_tradfac_leader }
				}
			}
		}
		pop_faction_event = { id = factions.1 } # call after to make sure leader assignment makes sense
	}

	on_destroy = {
		if = {
			limit = {
				has_modifier = lrsk_gvp_promoted_ritualist
			}
			remove_modifier = lrsk_gvp_promoted_ritualist
		}
		if = {
			limit = {
				has_modifier = lrsk_gvp_suppressed_ritualist
			}
			remove_modifier = lrsk_gvp_suppressed_ritualist
		}
	}

	actions = {
		embrace_faction = {
			title = "EMBRACE_FACTION"
			description = "EMBRACE_FACTION_DESC"

			cost = {
				influence = 500
			}

			potential = {
				exists = owner
				owner = {
					OR = {
						is_subject = no
						NOT = { is_subject_type = dominion }
					}
				}
			}

			valid = {
				custom_tooltip = {
					fail_text = EMBRACE_FACTION_COOLDOWN
					parameter:empire = {
						NOT = { has_modifier = embraced_faction_timer }
					}
				}
				support > 0.20
				parameter:empire = {
					NOT = { has_ethic = "ethic_fanatic_spiritualist" }
				}
			}

			effect = {
				add_modifier = { modifier = embraced_faction days = 3600 }
				parameter:empire = {
					shift_ethic = ethic_spiritualist
					hidden_effect = {
						add_modifier = { modifier = embraced_faction_timer days = 3600 }
						every_pop_faction = {
							limit = { NOT = { is_same_value = root } }
							add_modifier = { modifier = embraced_another_faction days = 3600 }
						}
					}
				}
				hidden_effect = {
					save_event_target_as = TargetFaction
					parameter:empire = {
						every_relation = {
							limit = {
								is_ai = no
								is_country_type = default
								has_communications = prev
								has_intel_level = {
									who = prev
									category = government
									level >= 1
								}
							}
							country_event = { id = factions.2000 }
						}
					}
				}
			}

			ai_weight = {
				base = 1
				modifier = {
					factor = 0
					support < 0.50
					owner = {
						has_ethic = ethic_spiritualist
					}
				}
				modifier = {
					factor = 0
					NOT = {
						owner = {
							exists = ruler
							ruler = {
								leader_of_faction = gvp_tb_ritualist
							}
						}
					}
				}
				modifier = {
					factor = 0
					owner = {
						count_pop_faction = {
							count < 4
						}
					}
				}
			}
		}
		promote_faction = {
			title = "PROMOTE_FACTION"
			description = "PROMOTE_FACTION_DESC"

			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = lrsk_gvp_suppressed_ritualist
						has_modifier = lrsk_gvp_promoted_ritualist
					}
				}
			}

			effect = {
				parameter:empire = {
					add_modifier = { modifier = lrsk_gvp_promoted_ritualist days = -1 }
				}
			}

			ai_weight = {
				base = 0
			}
		}
		cancel_promote_faction = {
			title = "CANCEL_PROMOTE_FACTION"
			description = "CANCEL_PROMOTE_FACTION_DESC"

			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = lrsk_gvp_promoted_ritualist
				}
			}

			effect = {
				parameter:empire = {
					remove_modifier = lrsk_gvp_promoted_ritualist
				}
			}

			ai_weight = {
				base = 0
			}
		}
		suppress_faction = {
			title = "SUPPRESS_FACTION"
			description = "GVP_TB_RITUALIST_SUPPRESS_DESC"

			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = lrsk_gvp_suppressed_ritualist
						has_modifier = lrsk_gvp_promoted_ritualist
					}
				}
			}

			effect = {
				add_modifier = { modifier = suppressed_faction days = -1 }
				parameter:empire = {
					add_modifier = { modifier = lrsk_gvp_suppressed_ritualist days = -1 }
				}
			}

			ai_weight = {
				base = 0
			}
		}
		cancel_suppress_faction = {
			title = "CANCEL_SUPPRESS_FACTION"
			description = "CANCEL_SUPPRESS_FACTION_DESC"

			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = lrsk_gvp_suppressed_ritualist
				}
			}

			effect = {
				remove_modifier = suppressed_faction
				parameter:empire = {
					remove_modifier = lrsk_gvp_suppressed_ritualist
				}
			}

			ai_weight = {
				base = 0
			}
		}
	}
