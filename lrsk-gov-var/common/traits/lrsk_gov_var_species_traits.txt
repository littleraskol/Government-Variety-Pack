## Shroud Touched

lrsk_gov_var_trait_natural_esper = {
	cost = 0
	icon = "gfx/interface/icons/traits/lrsk_gov_var_trait_natural_esper.dds"

	improves_leaders = yes	
	initial = no
	randomized = no
	species_potential_add = {
		always = no
	}
	species_possible_remove = {
		always = no
	}
	species_possible_merge_add = {
		always = no
	}

	allowed_archetypes = { BIOLOGICAL LITHOID }
	tags = { organic positive pop_output research unity happiness psionic special }

	modifier = {
		researcher_jobs_bonus_workforce_mult = 0.025
		bureaucrat_jobs_bonus_workforce_mult = 0.025
		telepath_jobs_bonus_workforce_mult = 0.025
	}

	slave_cost = {
		energy = 750
	}

	assembly_score = {
		base = 0.5
		modifier = {
			add = 0.5
			from = { has_research_designation = yes }
		}
		modifier = {
			add = 0.5
			from = { has_unity_designation = yes }
		}
		modifier = {
			subtract = 1
			exists = from.owner
			from.owner = { has_origin = lrsk_gov_var_origin_shroud_touched }
		}
	}
}
