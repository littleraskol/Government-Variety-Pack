lrsk_gvp_leader_trait_esper = {
	is_councilor_trait_for_ruler = yes
	inline_script = {
		script = trait/icon_lrsk_gvp_esper
		CLASS = leader
		ICON = "GFX_lrsk_gvp_leader_trait_esper"
		RARITY = common
		COUNCIL = triggered
		TIER = none
	}
	triggered_modifier = {
		potential = {
			is_ruler = no
			leader_class = admiral
		}
		ship_weapon_damage = 0.05
		ship_shield_hardening_add = 0.05
	}
	triggered_modifier = {
		potential = {
			is_ruler = no
			leader_class = general
		}
		army_morale = 0.1
		army_morale_damage_mult = 0.1
	}
	triggered_modifier = {
		potential = {
			is_ruler = no
			leader_class = scientist
		}
		science_ship_survey_speed = 0.15
		ship_anomaly_generation_chance_mult = 0.1
	}
	triggered_self_modifier = {
		potential = {
			is_ruler = no
			leader_class = scientist
		}
		leader_scientists_physics_research_produces_add = 1.5
		leader_scientists_society_research_produces_add = 2
		leader_scientists_engineering_research_produces_add = 1.5
		mult = trigger:has_skill
	}
	triggered_modifier = {
		potential = {
			is_ruler = no
			leader_class = governor
		}
		planet_jobs_unity_produces_mult = 0.05
	}
	triggered_modifier = {
		potential = {
			is_ruler = yes
		}
		pop_government_ethic_attraction = 0.05
		country_ruler_influence_produces_add = 0.25
		category_psionics_research_speed_mult = 0.05
	}
	leader_class = { admiral scientist general governor }
	opposites = {
		leader_trait_cyborg
		leader_trait_synthetic
		leader_trait_psionic
	}
	initial = no
	randomized = no
	notify_on_gained = no
}
