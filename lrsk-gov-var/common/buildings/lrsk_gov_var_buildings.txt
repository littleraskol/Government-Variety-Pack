@lrsk_traders_from_civic = 100
@lrsk_trade_value_from_jobs = 0.05
@lrsk_trade_value_from_jobs_small = 0.025

# Managerial Technique Institute
lrsk_gvp_bldg_managerial_technique_institute = {
	base_buildtime = @b1_time
	base_cap_amount = 1

	owner_type = corporate
	category = trade

	potential = {
		has_branch_office = yes
		branch_office_owner = { has_valid_civic = lrsk_gov_var_civic_mc_sci_mgnt }
	}

	planet_modifier = {
		job_lrsk_gov_var_job_functionary_add = 200
	}
	
	triggered_desc = {
		trigger = {
			branch_office_owner = { is_legit_or_legal_criminal = yes }
		}
		text = lrsk_legit_mgmt_bonus
	}

	triggered_planet_modifier = {
		potential = {
			exists = branch_office_owner
			branch_office_owner = { is_legit_or_legal_criminal = yes }
		}
		modifier = {
			branch_office_value_mult = 0.1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = branch_office_owner
			branch_office_owner = { is_legit_or_legal_criminal = yes }
			num_assigned_jobs = {
				job = lrsk_gov_var_job_functionary
				value > 0
			}
		}
		modifier = {
			branch_office_value_mult = 0.05
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = branch_office_owner
			branch_office_owner = { is_legit_or_legal_criminal = yes }
			num_assigned_jobs = {
				job = lrsk_gov_var_job_functionary
				value >= 100
			}
		}
		modifier = {
			branch_office_value_mult = 0.1
		}
	}
	
	triggered_desc = {
		trigger = {
			branch_office_owner = { is_illegal_criminal = yes }
		}
		text = lrsk_crim_mgmt_bonus
	}

	triggered_planet_modifier = {
		potential = {
			exists = branch_office_owner
			branch_office_owner = { is_illegal_criminal = yes }
		}
		modifier = {
			planet_crime_add = 20
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = branch_office_owner
			branch_office_owner = { is_illegal_criminal = yes }
			num_assigned_jobs = {
				job = lrsk_gov_var_job_functionary
				value > 0
			}
		}
		modifier = {			
			planet_crime_mult = 0.1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = branch_office_owner
			branch_office_owner = { is_illegal_criminal = yes }
			num_assigned_jobs = {
				job = lrsk_gov_var_job_functionary
				value >= 100
			}
		}
		modifier = {			
			planet_crime_mult = 0.2
		}
	}

	triggered_planet_modifier = {
		potential = {
			always = yes
		}
		planet_jobs_trade_produces_mult = @lrsk_trade_value_from_jobs_small
		mult = value:num_jobs_with_tag|TAG|trade|
	}
	triggered_planet_modifier = {
		potential = {
			always = yes
		}
		planet_jobs_trade_produces_mult = @lrsk_trade_value_from_jobs_small
		mult = value:num_jobs_with_tag|TAG|industry|
	}
	#Merchant Jobs (normalizing with base game standards)
	triggered_country_modifier = {
		potential = {
			always = yes
		}
		job_capital_trader_add = @lrsk_traders_from_civic
	}

	resources = {
		category = planet_branch_office_buildings
		cost = {
			minerals = 500
			influence = 10
		}
		cost = {
			trigger = {
				owner = {
					has_auth_corporate = yes
				}
			}
			influence = 10
		}
		upkeep = {
			energy = 2
		}
	}

	ai_weight = {
		weight = 100
	}
}
