triggered_opinion_lrsk_adv_civ_rampant_ascendants = {
	trigger = {
		OR = {
			is_country_type = fallen_empire
			is_country_type = awakened_fallen_empire
			has_country_flag = fallen_empire_machine
			is_country_type = enclave
			relative_power = {
				who = from
				category = all
				value > equivalent
			}
		}
		FROM = {
			has_valid_civic = lrsk_adv_civ_civic_rampant_ascendants
		}
	}
	opinion = {
		base = -50
	}
}

triggered_opinion_lrsk_adv_civ_audacity_war = {

		}
		FROM = {
			has_edict = lrsk_adv_civ_audacity_war_edict
		}
	}
	opinion = {
		base = -100
	}
}

triggered_opinion_lrsk_adv_civ_const_monarch_like = {
	trigger = {
		OR = {
			has_authority = auth_imperial
			AND = {
				has_authority = auth_dictatorial
				has_valid_civic = civic_philosopher_king
			}
		}
		FROM = {
			lrsk_adv_civ_has_const_monarch_civic = yes
		}
	}
	opinion = {
		base = 25
	}
}

triggered_opinion_lrsk_adv_civ_const_monarch_dislike = {
	trigger = {
		has_ethic = ethic_fanatic_egalitarian
		FROM = {
			lrsk_adv_civ_has_const_monarch_civic = yes
		}
	}
	opinion = {
		base = -10
	}
}
